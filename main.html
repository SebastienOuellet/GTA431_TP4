def obtenir_gravite_accident(donnees_accident):

# Permet d'obtenir le titre de chaque titre du fichier csv cr manuellement
def obtenir_titre_genre():
    #dictionnaire genre avec cl code : valeur= genre
    dictionnaire_genre = {}  
    donnees_genre = pd.read_csv("genre.csv", delimiter=",")
    # Cration du dictionnaire
    for index, row in donnees_genre.iterrows():
        if row['genre'] in dictionnaire_genre.keys():
            pass
        else:
            dictionnaire_genre[row['genre']] = row["titre"]
    return dictionnaire_genre 
       
# Fonction pour obtenir et afficher le tableau des genre d'accident
def obtenir_genre_accident(donnees_accident):
    dictionnaire_genre = obtenir_titre_genre();
    #Obtenir le nombre d'accident par genre des donnes extraite du csv (Retourne type et total)
    data_genre_accident = donnees_accident["CD_GENRE_ACCDN"].value_counts()
    
    #Pour chaque donnes (donnees_accident) du fichier excel avec un genre Int, on va chercher son titre avec le dictionnaire de genre
    titre = []
    total_accident = []
    for genre, total in data_genre_accident.iteritems():
        if genre in dictionnaire_genre.keys():
            titre.append(dictionnaire_genre[genre])
            total_accident.append(total)
    
    # Cration d'un dataframe pour faire l'affichage du tableau 
    pd_genre_accident = pd.DataFrame({ 'Genre':titre, 'Total': total_accident})
    
    #cration du diagrame a bar
    ax = pd_genre_accident.plot(kind="barh", figsize=(20,15), color="#ff5722", fontsize=14)
    
    #Modification du titre, le nom des axes et la lgendes
    ax.set_title("Le nombre d'accidents en fonction du genre de ceux-ci depuis 2012", fontsize=18)
    plt.xlabel('Nombres d\'accidents', fontsize=18)
    plt.ylabel('Genre d\'accidents', fontsize=18)
    plt.legend(['Genre d\'accidents'])
    
    # Permet de mettre les genre en string pour remplac les type du fichier excel.
    ax.set_yticklabels(pd_genre_accident["Genre"],fontsize=14)

    #Ajout du nombre d'accident en haut de chaque genre pour bien voir le nombre de ceux-ci
    for index,data in enumerate(data_genre_accident):
        plt.text(x=data+1, y=index-0.2 , s=f"{data}" , fontdict=dict(fontsize=14))

# Permet d'obtenir l'extraction des donnes en pourcentage par annes
def afficher_repartition_accident_selon_annee(donnees_accident):
    annee = [2012,2013,2014,2015,2016,2017,2018]
    nombre_accident_par_annee = [
        donnees_accident[donnees_accident['AN'] == 2012].count()['AN'],
        donnees_accident[donnees_accident['AN'] == 2013].count()['AN'],
        donnees_accident[donnees_accident['AN'] == 2014].count()['AN'],
        donnees_accident[donnees_accident['AN'] == 2015].count()['AN'],
        donnees_accident[donnees_accident['AN'] == 2016].count()['AN'],
        donnees_accident[donnees_accident['AN'] == 2017].count()['AN'],
        donnees_accident[donnees_accident['AN'] == 2018].count()['AN'],        
    ]

    # Ex pris de https://medium.com/@kvnamipara/a-better-visualisation-of-pie-charts-by-matplotlib-935b7667d77f
    # Cration du Pie chart qui reprsente les rpartition des accidents selon les annes
    fig, ax = plt.subplots()
    ax.pie(nombre_accident_par_annee, labels=annee, autopct='%1.1f%%', shadow=False, startangle=95)
    # Equal aspect ratio ensures that pie is drawn as a circle
    ax.axis('equal')
    ax.set_title("Pourcentage du nombre d\'accident par annes'")
    plt.tight_layout()

# Permet d'obtenir le tableau agrg des accidents et du nombre de victime
def obtenir_tableau_agrege(donnees_accident):
    donne_analyse = donnees_accident[['GRAVITE','AN','NO_SEQ_COLL', 'NB_VICTIMES_TOTAL']]
    # NO_SEQ_COLL est le ID unique, il servira dans le calcul de la fonction d'aggrgation pour ne pas avoir de doublons
    # AN les annes seront nos columns, C--D que les donnes seront regroupes par annes
    # GRAVITE la gravit des accidents sont les donnes que nous voulons reprsent, par anne
    tableau_aggrege =\
        pd.pivot_table(donne_analyse,
                       values=['NO_SEQ_COLL'],
                       index=['GRAVITE', 'NB_VICTIMES_TOTAL'],
                       columns=['AN'],
                       aggfunc={'GRAVITE': 'count'}, fill_value=0
                       )
    #cration du tableau aggrg, les valeurs NO_SEQ_COLL pour chaque type de GRAVIT sont compter, puis regrouper par anne
    return tableau_aggrege
